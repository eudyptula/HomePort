101  hpd_adapter_t *adapter;
102  hpd_device_t *device;
103  HPD_TAILQ_FOREACH(adapter, &configuration->adapters) {

188 hpd_error_t event_changed(hpd_service_id_t *id, hpd_value_t *val)
189 {
190  hpd_error_t rc = HPD_E_ALLOC;
191  hpd_ev_async_t *async;
192  HPD_CALLOC(async, 1, hpd_ev_async_t);
193  HPD_CPY_ALLOC(async->value, val, hpd_value_t);
194  if ((rc = discovery_copy_sid(&async->service, id))) goto copy_sid_error;
195  hpd_t *hpd = id->device.adapter.hpd;
196  async->hpd = hpd;
197  ev_async_init(&async->watcher, on_changed);
198  async->watcher.data = async;
199  ev_async_start(hpd->loop, &async->watcher);
200  ev_async_send(hpd->loop, &async->watcher);
201  TAILQ_INSERT_TAIL(&hpd->changed_watchers, async, HPD_TAILQ_FIELD);
202  free(val);
203  return HPD_E_SUCCESS;
204 
205  copy_sid_error:
206  free(async->value);
207  alloc_error:
208  free(async);
209  switch (rc) {
210  case HPD_E_ALLOC:
211  LOG_RETURN_E_ALLOC();
212  default:
213  return rc;
214  }
215 }
