122 static hpd_error_t add_service(json_t *parent, hpd_service_id_t *service, hpd_rest_t *rest, hpd_module_t *context)
123 {
124  hpd_error_t rc;
125 
126  // Create object
127  json_t *json;
128  if (!(json = json_object())) RETURN_JSON_ERROR(context);
129 
130  // Add id
131  const char *id;
132  if ((rc = hpd_service_get_id(service, &id))) goto error;
133  if ((rc = add(json, HPD_REST_KEY_ID, id, context))) goto error;
134 
135  // Add url
136  char *url;
137  if ((rc = hpd_rest_url_create(rest, service, &url))) goto error;
138  if ((rc = add(json, HPD_REST_KEY_URI, url, context))) {
139  free(url);
140  goto error;
141  }
142  free(url);
143 
144  // Add actions
145  hpd_action_t *action;
146  hpd_service_foreach_action(rc, action, service) {
147  hpd_method_t method;
148  if ((rc = hpd_action_get_method(action, &method))) goto error;
149  switch (method) {
150  case HPD_M_NONE:break;
151  case HPD_M_GET:
152  if ((rc = add(json, HPD_REST_KEY_GET, HPD_REST_VAL_TRUE, context))) goto error;
153  break;
154  case HPD_M_PUT:
155  if ((rc = add(json, HPD_REST_KEY_PUT, HPD_REST_VAL_TRUE, context))) goto error;
156  break;
157  case HPD_M_COUNT:break;
158  }
159  }
160  if (rc) goto error;
161 
162  // Add attributes
163  hpd_pair_t *pair;
164  hpd_service_foreach_attr(rc, pair, service)
165  if ((rc = add_attr(json, pair, context))) goto error;
166  if (rc) goto error;
167 
168  // Add parameters
169  if ((rc = add_parameters(json, service, context))) goto error;
170 
171  // Add to parent
172  if (json_array_append_new(parent, json)) {
173  json_decref(json);
174  RETURN_JSON_ERROR(context);
175  }
176 
177  return HPD_E_SUCCESS;
178 
179  error:
180  json_decref(json);
181  return rc;
182 }

338 static hpd_error_t add_configuration(json_t *parent, hpd_t *hpd, hpd_rest_t *rest, hpd_module_t *context)
339 {
340  hpd_error_t rc;
341 
342  // Create object
343  json_t *json;
344  if (!(json = json_object())) RETURN_JSON_ERROR(context);
345 
346  // Add encoded charset
347 #ifdef CURL_ICONV_CODESET_OF_HOST
348  curl_version_info_data *curl_ver = curl_version_info(CURLVERSION_NOW);
349  if (curl_ver->features & CURL_VERSION_CONV && curl_ver->iconv_ver_num != 0)
350  if ((rc = add(json, HPD_REST_KEY_URL_ENCODED_CHARSET, CURL_ICONV_CODESET_OF_HOST, context))) goto error;
351  else
352  if ((rc = add(json, HPD_REST_KEY_URL_ENCODED_CHARSET, HPD_REST_VAL_ASCII, context))) goto error;
353 #else
354  if ((rc = add(json, HPD_REST_KEY_URL_ENCODED_CHARSET, HPD_REST_VAL_ASCII, context))) goto error;
355 #endif
356 
357  // Add adapters
358  if ((rc = add_adapters(json, hpd, rest, context))) goto error;
359 
360  // Add to parent
361  if (json_object_set_new(parent, HPD_REST_KEY_CONFIGURATION, json)) {
362  json_decref(json);
363  RETURN_JSON_ERROR(context);
364  }
365 
366  return HPD_E_SUCCESS;
367 
368  error:
369  json_decref(json);

423 hpd_error_t hpd_rest_json_parse_value(const char *in, hpd_module_t *context, char **out)
424 {
425  // Load json
426  json_t *json = NULL;
427  json_error_t *error = NULL;
428  if (!(json = json_loads(in, 0, error))) {
429  HPD_LOG_RETURN(context, HPD_E_ARGUMENT, "Json parsing error: %s", error->text);
430  }
431 
432  // Get value
433  json_t *value;
434  if (!json_is_object(json) || !(value = json_object_get(json, HPD_REST_KEY_VALUE))) {
435  json_decref(json);
436  HPD_LOG_RETURN(context, HPD_E_ARGUMENT, "Json parsing error");
437  }
438 
439  // Allocate a copy
440  (*out) = malloc((json_string_length(value)+1)*sizeof(char));
441  if (!(*out)) {
442  json_decref(json);
443  HPD_LOG_RETURN_E_ALLOC(context);
444  }
445  strcpy((*out), json_string_value(value));
446 
447  json_decref(json);
448  return HPD_E_SUCCESS;
449 }
