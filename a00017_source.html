114 hpd_error_t hpd_httpd_response_create(hpd_httpd_response_t **response, hpd_httpd_request_t *req, hpd_status_t status)
115 {
116  if (!req) return HPD_E_NULL;
117  hpd_error_t rc, rc2;
118  hpd_module_t *context;
119  if ((rc = http_request_get_context(req, &context))) return rc;
120  if (!response) HPD_LOG_RETURN_E_NULL(context);
121 
122  int len;
123 
124  // Get data
125  char *status_str = http_status_codes_to_str(status);
126 
127  // Calculate msg length
128  len = 1;
129  len += strlen(HTTP_VERSION);
130  if (status_str) len += strlen(status_str);
131  len += strlen(CRLF);
132 
133  // Allocate space
134  (*response) = malloc(sizeof(hpd_httpd_response_t));
135  if (!(*response)) HPD_LOG_RETURN_E_ALLOC(context);
136  (*response)->status = status;
137  (*response)->msg = malloc(len*sizeof(char));
138  if (!(*response)->msg) {
139  if ((rc = hpd_httpd_response_destroy((*response))))
140  HPD_LOG_ERROR(context, "Failed to destroy response (code: %d).", rc);
141  HPD_LOG_RETURN_E_ALLOC(context);
142  }
143 
144  // Init struct
145  (*response)->context = context;
146  if ((rc = http_request_get_connection(req, &(*response)->conn))) return rc;
147 
148  // Construct msg
149  strcpy((*response)->msg, HTTP_VERSION);
150  if (status_str) strcat((*response)->msg, status_str);
151  strcat((*response)->msg, CRLF);
152 
153  // Real persistent connections is not supported, so tell client that we close connection after response has been sent
154  rc = hpd_httpd_response_add_header((*response), "Connection", "close");
155  if (rc && (rc2 = hpd_httpd_response_destroy(*response)))
156  HPD_LOG_ERROR(context, "Failed to destroy response (code: %d).", rc2);;
157  return rc;
158 }

174 hpd_error_t hpd_httpd_response_add_header(hpd_httpd_response_t *res, const char *field, const char *value)
175 {
176  if (!res) return HPD_E_NULL;
177  if (!field || !value) HPD_LOG_RETURN_E_NULL(res->context);
178 
179  // Headers already sent
180  if (!res->msg)
181  HPD_LOG_RETURN(res->context, HPD_E_STATE, "Cannot add header, they have already been sent to client.");
182 
183  char *msg;
184  size_t msg_len = strlen(res->msg)+strlen(field)+2+strlen(value)+strlen(CRLF)+1;
185 
186  msg = realloc(res->msg, msg_len*sizeof(char));
187  if (!msg) HPD_LOG_RETURN_E_ALLOC(res->context);
188  res->msg = msg;
189 
190  strcat(res->msg, field);
191  strcat(res->msg, ": ");
192  strcat(res->msg, value);
193  strcat(res->msg, CRLF);
194 
195  return HPD_E_SUCCESS;
196 }

218 hpd_error_t hpd_httpd_response_add_cookie(hpd_httpd_response_t *res,
219  const char *field, const char *value,
220  const char *expires, const char *max_age,
221  const char *domain, const char *path,
222  int secure, int http_only,
223  const char *extension)
224 {
225  if (!res) return HPD_E_NULL;
226  if (!field || !value) HPD_LOG_RETURN_E_NULL(res->context);
227 
228  // Headers already sent
229  if (!res->msg) return HPD_E_STATE;
230 
231  char *msg;
232  size_t msg_len = strlen(res->msg) + 12 + strlen(field) + 1 + strlen(value) + strlen(CRLF) + 1;
233 
234  // Calculate length
235  if (expires) msg_len += 10 + strlen(expires);
236  if (max_age) msg_len += 10 + strlen(max_age);
237  if (domain) msg_len += 9 + strlen(domain);
238  if (path) msg_len += 7 + strlen(path);
239  if (secure) msg_len += 8;
240  if (http_only) msg_len += 10;
241  if (extension) msg_len += 2 + strlen(extension);
242 
243  // Reallocate message
244  msg = realloc(res->msg, msg_len*sizeof(char));
245  if (!msg) HPD_LOG_RETURN_E_ALLOC(res->context);
246  res->msg = msg;
247 
248  // Apply header
249  strcat(res->msg, "Set-Cookie: ");
250  strcat(res->msg, field);
251  strcat(res->msg, "=");
252  strcat(res->msg, value);
253  strcat(res->msg, CRLF);
254  if (expires) {
255  strcat(res->msg, "; Expires=");
256  strcat(res->msg, expires);
257  }
258  if (max_age) {
259  strcat(res->msg, "; Max-Age=");
260  strcat(res->msg, max_age);
261  }
262  if (domain) {
263  strcat(res->msg, "; Domain=");
264  strcat(res->msg, domain);
265  }
266  if (path) {
267  strcat(res->msg, "; Domain=");
268  strcat(res->msg, domain);
269  }
270  if (secure) strcat(res->msg, "; Secure");
271  if (http_only) strcat(res->msg, "; HttpOnly");
272  if (extension) {
273  strcat(res->msg, "; ");
274  strcat(res->msg, extension);
275  }
276 
277  return HPD_E_SUCCESS;
278 }

317 hpd_error_t hpd_httpd_response_vsendf(hpd_httpd_response_t *res, const char *fmt, va_list arg)
318 {
319  if (!res) return HPD_E_NULL;
320 
321  hpd_error_t rc;
322 
323  if (res->msg) {
324  const char *ip;
325  if ((rc = hpd_tcpd_conn_get_ip(res->conn, &ip))) {
326  HPD_LOG_WARN(res->context, "Failed to get ip.");
327  ip = "(unknown)";
328  }
329  HPD_LOG_VERBOSE(res->context, "Sending response to %s: %i %s.", ip, res->status, http_status_codes_to_str(res->status));
330  if ((rc = hpd_tcpd_conn_sendf(res->conn, "%s%s", res->msg, CRLF))) return rc;
331  free(res->msg);
332  res->msg = NULL;
333  }
334 
335  if (fmt) {
336  return hpd_tcpd_conn_vsendf(res->conn, fmt, arg);
337  }
338 
339  return HPD_E_SUCCESS;
340 }
