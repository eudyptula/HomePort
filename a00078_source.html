79 static hpd_error_t add_service(mxml_node_t *parent, hpd_service_id_t *service, hpd_rest_t *rest, hpd_module_t *context)
80 {
81  hpd_error_t rc;
82 
83  // Create node
84  mxml_node_t *xml;
85  if (!(xml = mxmlNewElement(parent, HPD_REST_KEY_SERVICE))) RETURN_XML_ERROR(context);
86 
87  // Add id
88  const char *id;
89  if ((rc = hpd_service_get_id(service, &id))) return rc;
90  if ((rc = add(xml, HPD_REST_KEY_ID, id, context))) return rc;
91 
92  // Add url
93  char *url;
94  if ((rc = hpd_rest_url_create(rest, service, &url))) return rc;
95  if ((rc = add(xml, HPD_REST_KEY_URI, url, context))) {
96  free(url);
97  return rc;
98  }
99  free(url);
100 
101  // Add actions
102  hpd_action_t *action;
103  hpd_service_foreach_action(rc, action, service) {
104  hpd_method_t method;
105  if ((rc = hpd_action_get_method(action, &method))) return rc;
106  switch (method) {
107  case HPD_M_NONE:break;
108  case HPD_M_GET:
109  if ((rc = add(xml, HPD_REST_KEY_GET, HPD_REST_VAL_TRUE, context))) return rc;
110  break;
111  case HPD_M_PUT:
112  if ((rc = add(xml, HPD_REST_KEY_PUT, HPD_REST_VAL_TRUE, context))) return rc;
113  break;
114  case HPD_M_COUNT:break;
115  }
116  }
117  if (rc) return rc;
118 
119  // Add attributes
120  hpd_pair_t *pair;
121  hpd_service_foreach_attr(rc, pair, service)
122  if ((rc = add_attr(xml, pair, context))) return rc;
123  if (rc) return rc;
124 
125  // Add parameters
126  hpd_parameter_id_t *parameter;
127  hpd_service_foreach_parameter(rc, parameter, service) {
128  if ((rc = add_parameter(xml, parameter, context))) return rc;
129  }
130  if (rc) return rc;
131 
132  return HPD_E_SUCCESS;
133 }

194 static hpd_error_t add_configuration(mxml_node_t *parent, hpd_t *hpd, hpd_rest_t *rest, hpd_module_t *context)
195 {
196  hpd_error_t rc;
197 
198  // Create object
199  mxml_node_t *xml;
200  if (!(xml = mxmlNewElement(parent, HPD_REST_KEY_CONFIGURATION))) RETURN_XML_ERROR(context);
201 
202  // Add encoded charset
203 #ifdef CURL_ICONV_CODESET_OF_HOST
204  curl_version_info_data *curl_ver = curl_version_info(CURLVERSION_NOW);
205  if (curl_ver->features & CURL_VERSION_CONV && curl_ver->iconv_ver_num != 0)
206  if ((rc = add(xml, HPD_REST_KEY_URL_ENCODED_CHARSET, CURL_ICONV_CODESET_OF_HOST, context))) return rc;
207  else
208  if ((rc = add(xml, HPD_REST_KEY_URL_ENCODED_CHARSET, HPD_REST_VAL_ASCII, context))) return rc;
209 #else
210  if ((rc = add(xml, HPD_REST_KEY_URL_ENCODED_CHARSET, HPD_REST_VAL_ASCII, context))) return rc;
211 #endif
212 
213  // Add adapters
214  hpd_adapter_id_t *adapter;
215  hpd_foreach_adapter(rc, adapter, hpd) {
216  if ((rc = add_adapter(xml, adapter, rest, context))) return rc;
217  }
218  if (rc) return rc;
219 
220  return HPD_E_SUCCESS;
221 }

317 hpd_error_t hpd_rest_xml_parse_value(const char *in, hpd_module_t *context, char **out)
318 {
319  BEGIN(context);
320 
321  // TODO Return HPD_E_ARGUMENT on parse errors!
322 
323  mxml_node_t *xml;
324  if (!(xml = mxmlLoadString(NULL, in, MXML_TEXT_CALLBACK))) {
325  END();
326  HPD_LOG_RETURN(context, HPD_E_ARGUMENT, "XML parsing error");
327  }
328 
329  mxml_node_t *node;
330  node = mxmlFindElement(xml, xml, HPD_REST_KEY_VALUE, NULL, NULL, MXML_DESCEND);
331  if (!node) {
332  mxmlDelete(xml);
333  END();
334  HPD_LOG_RETURN(context, HPD_E_ARGUMENT, "XML parsing error");
335  }
336 
337  // TODO Alloc checks !
338  size_t len = 1;
339  (*out) = calloc(len, sizeof(char));
340  for (node = node->child; node && node->type == MXML_TEXT; node = node->next) {
341  int whitespaces = node->value.text.whitespace;
342  char *string = node->value.text.string;
343  (*out) = realloc((*out), (len + whitespaces + strlen(string)) * sizeof(char));
344  for (int i = 0; i < whitespaces; i++) strcat(*out, " ");
345  strcat(*out, string);
346  }
347 
348  mxmlDelete(xml);
349  END();
350  return HPD_E_SUCCESS;
