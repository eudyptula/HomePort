46 hpd_error_t curl_ev_init(curl_ev_handle_t **handle, hpd_module_t *context)
47 {
48  if (!context) return HPD_E_NULL;
49  if (!handle) HPD_LOG_RETURN_E_NULL(context);
50 
51  CURLcode cc;
52 
53  HPD_CALLOC(*handle, 1, curl_ev_handle_t);
54 
55  (*handle)->context = context;
56  if (!((*handle)->handle = curl_easy_init())) goto init_error;
57  if ((cc = curl_easy_setopt((*handle)->handle, CURLOPT_HEADERFUNCTION, on_header))) goto curl_error;
58  if ((cc = curl_easy_setopt((*handle)->handle, CURLOPT_HEADERDATA, *handle))) goto curl_error;
59  if ((cc = curl_easy_setopt((*handle)->handle, CURLOPT_WRITEFUNCTION, on_body))) goto curl_error;
60  if ((cc = curl_easy_setopt((*handle)->handle, CURLOPT_WRITEDATA, *handle))) goto curl_error;
61 
62  return HPD_E_SUCCESS;
63 
64  curl_error:
65  curl_easy_cleanup((*handle)->handle);
66  free(handle);
67  HPD_LOG_RETURN(context, HPD_E_UNKNOWN, "Curl returned an error [code: %i]", cc);
68 
69  init_error:
70  free(handle);
71  HPD_LOG_RETURN(context, HPD_E_UNKNOWN, "Curl init error");
72 
73  alloc_error:
74  HPD_LOG_RETURN_E_ALLOC(context);
75 }

77 hpd_error_t curl_ev_cleanup(curl_ev_handle_t *handle)
78 {
79  if (!handle) return HPD_E_NULL;
80  if (handle->curl_ev) HPD_LOG_RETURN(handle->context, HPD_E_STATE, "Handle is still attached");
81 
82  if (handle->headers) curl_slist_free_all(handle->headers);
83  if (handle->handle) curl_easy_cleanup(handle->handle);
84  if (handle->on_free) handle->on_free(handle->data);
85  free(handle);
86 
87  return HPD_E_SUCCESS;
88 }
