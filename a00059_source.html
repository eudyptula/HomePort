35 hpd_error_t discovery_alloc_adapter(hpd_adapter_t **adapter, const char *id)
36 {
37  hpd_error_t rc;
38 
39  HPD_CALLOC((*adapter), 1, hpd_adapter_t);
40  HPD_CALLOC((*adapter)->devices, 1, devices_t);
41  TAILQ_INIT((*adapter)->devices);
42  if ((rc = hpd_map_alloc(&(*adapter)->attributes))) {
43  discovery_free_adapter(*adapter);
44  (*adapter) = NULL;
45  return rc;
46  }
47  HPD_STR_CPY((*adapter)->id, id);
48  return HPD_E_SUCCESS;
49 
50  alloc_error:
51  if (*adapter) discovery_free_adapter(*adapter);
52  (*adapter) = NULL;
53  LOG_RETURN_E_ALLOC();
54 }

185 hpd_error_t discovery_attach_adapter(hpd_t *hpd, hpd_adapter_t *adapter)
186 {
187  hpd_error_t rc;
188  hpd_adapter_t *copy;
189  HPD_CPY_ALLOC(copy, adapter, hpd_adapter_t);
190 
191  TAILQ_INSERT_TAIL(&hpd->configuration->adapters, copy, HPD_TAILQ_FIELD);
192  copy->configuration = hpd->configuration;
193 
194  if ((rc = event_inform_adapter_attached(copy))) {
195  TAILQ_REMOVE(&hpd->configuration->adapters, copy, HPD_TAILQ_FIELD);
196  free(copy);
197  return rc;
198  }
199 
200  hpd_device_t *device;
201  HPD_TAILQ_FOREACH(device, adapter->devices) {
202  device->adapter = copy;
203  }
204 
205  free(adapter);
206  return HPD_E_SUCCESS;
207 
208  alloc_error:
209  LOG_RETURN_E_ALLOC();
210 }
