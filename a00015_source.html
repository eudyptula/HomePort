135 hpd_error_t hpd_httpd_create(hpd_httpd_t **httpd, hpd_httpd_settings_t *settings, hpd_module_t *context, hpd_ev_loop_t *loop)
136 {
137  if (!context) return HPD_E_NULL;
138  if (!httpd || !settings || !loop) HPD_LOG_RETURN_E_NULL(context);
139 
140  // Allocate instance
141  (*httpd) = malloc(sizeof(hpd_httpd_t));
142  if (!(*httpd)) HPD_LOG_RETURN_E_ALLOC(context);
143 
144  // Set context
145  (*httpd)->context = context;
146 
147  // Copy settings
148  memcpy(&(*httpd)->settings, settings, sizeof(hpd_httpd_settings_t));
149 
150  // Construct settings for tcpd
151  hpd_tcpd_settings_t ws_settings = HPD_TCPD_SETTINGS_DEFAULT;
152  ws_settings.port = settings->port;
153  ws_settings.timeout = settings->timeout;
154  ws_settings.on_connect = on_connect;
155  ws_settings.on_receive = on_receive;
156  ws_settings.on_disconnect = on_disconnect;
157  ws_settings.tcpd_ctx = (*httpd);
158 
159  // Create tcpd
160  hpd_error_t rc;
161  if ((rc = hpd_tcpd_create(&(*httpd)->webserver, &ws_settings, context, loop))) {
162  hpd_httpd_destroy(*httpd);
163  return rc;
164  }
165 
166  return HPD_E_SUCCESS;
167 }
