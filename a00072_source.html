38 hpd_error_t request_alloc_request(hpd_request_t **request, hpd_service_id_t *id, hpd_method_t method, hpd_response_f on_response)
39 {
40  hpd_error_t rc = HPD_E_ALLOC;
41  HPD_CALLOC(*request, 1, hpd_request_t);
42  if ((rc = discovery_copy_sid(&(*request)->service, id))) goto alloc_error;
43  (*request)->method = method;
44  (*request)->on_response = on_response;
45  return HPD_E_SUCCESS;
46 
47  alloc_error:
48  request_free_request(*request);
49  switch (rc) {
50  case HPD_E_ALLOC:
51  LOG_RETURN_E_ALLOC();
52  default:
53  return rc;
54  }
55 }

254 hpd_error_t request_request(hpd_request_t *request)
255 {
256  hpd_ev_async_t *async;
257  HPD_CALLOC(async, 1, hpd_ev_async_t);
258  HPD_CPY_ALLOC(async->request, request, hpd_request_t);
259  ev_async_init(&async->watcher, on_request);
260  async->watcher.data = async;
261  hpd_t *hpd = request->service->device.adapter.hpd;
262  ev_async_start(hpd->loop, &async->watcher);
263  ev_async_send(hpd->loop, &async->watcher);
264  TAILQ_INSERT_TAIL(&hpd->request_watchers, async, HPD_TAILQ_FIELD);
265  free(request);
266  return HPD_E_SUCCESS;
267 
268  alloc_error:
269  free(async);
270  LOG_RETURN_E_ALLOC();
271 }
