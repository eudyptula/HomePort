52 hpd_error_t hpd_module(hpd_t *hpd, const char *id, hpd_module_def_t *module_def)
53 {
54  // TODO Reserve module names hpd and log? (maybe even create them as modules)
55  hpd_module_t *module;
56  if (!hpd || !id || !module_def) LOG_RETURN_E_NULL();
57  if (hpd->configuration) LOG_RETURN(HPD_E_STATE, "Cannot add module while hpd is running.");
58  if (strchr(id, '-')) LOG_RETURN(HPD_E_ARGUMENT, "Module ids may not contain '-'.");
59  HPD_TAILQ_FOREACH(module, &hpd->modules)
60  if (strcmp(module->id, id) == 0)
61  LOG_RETURN(HPD_E_NOT_UNIQUE, "Module ids must be unique.");
62  return daemon_add_module(hpd, id, module_def);
63 }

65 hpd_error_t hpd_module_add_option(hpd_module_t *context, const char *name, const char *arg, int flags, const char *doc)
66 {
67  if (!context || !name) LOG_RETURN_E_NULL();
68  hpd_t *hpd = context->hpd;
69  if (!hpd->options) LOG_RETURN(HPD_E_STATE, "Can only add options during on_create().");
70  size_t name_index = strlen(context->id)+1;
71  for (int i = 0; i < hpd->options_count; i++) {
72  if (name_index < strlen(hpd->options[i].name) && strcmp(&hpd->options[i].name[name_index], name) == 0)
73  LOG_RETURN(HPD_E_NOT_UNIQUE, "Option names must be unique within the module.");
74  }
75  return daemon_add_option(context, name, arg, flags, doc);
76 }
