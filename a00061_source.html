41 static int parse_opt(int key, char *arg, struct argp_state *state)
42 {
43  hpd_error_t rc;
44  hpd_t *hpd = state->input;
45 
46  if (key >= 0xff && (key - 0xff) < hpd->options_count) {
47  hpd_module_t *module = hpd->option2module[key-0xff];
48  const char *name = hpd->options[key - 0xff].name;
49  while ((name++)[0] != '-');
50  switch ((rc = module->def.on_parse_opt(module->data, name, arg))) {
51  case HPD_E_SUCCESS:
52  return 0;
53  case HPD_E_ARGUMENT:
54  LOG_DEBUG("Module '%s' did not recognise the option '%s'.", module->id, name);
55  return ARGP_ERR_UNKNOWN;
56  default:
57  return rc;
58  }
59  }
60 
61  return 0;
62 }

96  ev_signal_init(&(*hpd)->sigint_watcher, sig_cb, SIGINT);

137 hpd_error_t daemon_add_option(hpd_module_t *context, const char *name, const char *arg, int flags, const char *doc)
138 {
139  hpd_t *hpd = context->hpd;
140  argp_option_t option = { 0 };
141  argp_option_t empty = { 0 };
142  char *name_alloc, *arg_alloc = NULL, *doc_alloc = NULL;
143 
144  HPD_CALLOC(name_alloc, strlen(name)+strlen(context->id)+2, char);
145  strcpy(name_alloc, context->id);
146  strcat(name_alloc, "-");
147  strcat(name_alloc, name);
148  if (arg) HPD_STR_CPY(arg_alloc, arg);
149  if (doc) HPD_STR_CPY(doc_alloc, doc);
150  option.name = name_alloc;
151  option.key = 0xff + hpd->options_count;
152  option.arg = arg_alloc;
153  option.flags = flags;
154  option.doc = doc_alloc;
155 
156  HPD_REALLOC(hpd->options, hpd->options_count+2, argp_option_t);
157  HPD_REALLOC(hpd->option2module, hpd->options_count+1, hpd_module_t *);
158 
159  hpd->option2module[hpd->options_count] = context;
160  hpd->options[hpd->options_count] = option;
161  hpd->options[hpd->options_count+1] = empty;
162  hpd->options_count++;
163 
164  return HPD_E_SUCCESS;
165 
166  alloc_error:
167  if (name_alloc) free(name_alloc);
168  if (arg_alloc) free(arg_alloc);
169  if (doc_alloc) free(doc_alloc);
170  LOG_RETURN_E_ALLOC();
171 }

174 hpd_error_t daemon_start(hpd_t *hpd, int argc, char *argv[])
175 {
176  hpd_error_t rc;
177  hpd_module_t *module;
178 
179  // Allocate run-time and option memory
180  if ((rc = daemon_alloc_conf(&hpd->configuration, hpd))) goto return_error;
181  rc = HPD_E_ALLOC;
182  HPD_CALLOC(hpd->options, 1, argp_option_t);
183 
184  // Call on_create() on modules
185  HPD_TAILQ_FOREACH(module, &hpd->modules)
186  if ((rc = module->def.on_create(&module->data, module)) != HPD_E_SUCCESS)
187  goto module_create_error;
188 
189  // Parse options
190  struct argp argp = {hpd->options, parse_opt };
191  if (argp_parse(&argp, argc, argv, 0, 0, hpd)) {
192  LOG_DEBUG("Error while parsing arguments.");
193  rc = HPD_E_ARGUMENT;
194  goto arg_error;
195  }
196 
197  // Deallocate option memory
198  argp_option_t empty = { 0 };
199  for (argp_option_t *option = hpd->options; memcmp(option, &empty, sizeof(argp_option_t)); option++) {
200  free((void *) option->name);
201  free((void *) option->arg);
202  free((void *) option->doc);
203  }
204  free(hpd->options);
205  hpd->options = NULL;
206  free(hpd->option2module);
207  hpd->option2module = NULL;
208 
209  // Create event loop
210  hpd->loop = ev_loop_new(EVFLAG_AUTO);
211  if (!hpd->loop) {
212  rc = HPD_E_ALLOC;
213  goto ev_new_error;
214  }
215 
216  // Call on_start() on modules
217  HPD_TAILQ_FOREACH(module, &hpd->modules)
218  if ((rc = module->def.on_start(module->data, hpd)) != HPD_E_SUCCESS)
219  goto module_start_error;
220 
221  // Start signal watchers
222  ev_signal_start(hpd->loop, &hpd->sigint_watcher);
223  ev_signal_start(hpd->loop, &hpd->sigterm_watcher);
224 
225  // Run until we are stopped
226  // TODO Return value of ev_run can be used to run until all watchers have stopped (e.g. nice shutdown)
227  ev_run(hpd->loop, 0);
228 
229  // Stop signal watchers
230  ev_signal_stop(hpd->loop, &hpd->sigterm_watcher);
231  ev_signal_stop(hpd->loop, &hpd->sigint_watcher);
232 
233  // Call on_stop() on modules
234  TAILQ_FOREACH_REVERSE(module, &hpd->modules, modules, HPD_TAILQ_FIELD)
235  if ((rc = module->def.on_stop(module->data, hpd)) != HPD_E_SUCCESS)
236  goto module_stop_error;
237 
238  // Stop other watchers
239  hpd_ev_async_t *async, *async_tmp;
240  HPD_TAILQ_FOREACH_SAFE(async, &hpd->request_watchers, async_tmp) {
241  TAILQ_REMOVE(&hpd->request_watchers, async, HPD_TAILQ_FIELD);
242  ev_async_stop(hpd->loop, &async->watcher);
243  request_free_request(async->request); // TODO Handle error code
244  free(async);
245  }
246  HPD_TAILQ_FOREACH_SAFE(async, &hpd->respond_watchers, async_tmp) {
247  TAILQ_REMOVE(&hpd->respond_watchers, async, HPD_TAILQ_FIELD);
248  ev_async_stop(hpd->loop, &async->watcher);
249  request_free_response(async->response); // TODO Handle error code
250  free(async);
251  }
252  HPD_TAILQ_FOREACH_SAFE(async, &hpd->changed_watchers, async_tmp) {
253  TAILQ_REMOVE(&hpd->changed_watchers, async, HPD_TAILQ_FIELD);
254  ev_async_stop(hpd->loop, &async->watcher);
255  discovery_free_sid(async->service); // TODO Handle error code
256  value_free(async->value); // TODO Handle error code
257  free(async);
258  }
259  HPD_TAILQ_FOREACH_SAFE(async, &hpd->attached_watchers, async_tmp) {
260  TAILQ_REMOVE(&hpd->attached_watchers, async, HPD_TAILQ_FIELD);
261  ev_async_stop(hpd->loop, &async->watcher);
262  discovery_free_did(async->device); // TODO Handle error code
263  free(async);
264  }
265  HPD_TAILQ_FOREACH_SAFE(async, &hpd->detached_watchers, async_tmp) {
266  TAILQ_REMOVE(&hpd->detached_watchers, async, HPD_TAILQ_FIELD);
267  ev_async_stop(hpd->loop, &async->watcher);
268  discovery_free_did(async->device); // TODO Handle error code
269  free(async);
270  }
271 
272  // Destroy event loop
273  ev_loop_destroy(hpd->loop);
274  hpd->loop = NULL;
275 
276  // Call on_destroy on modules
277  TAILQ_FOREACH_REVERSE(module, &hpd->modules, modules, HPD_TAILQ_FIELD)
278  if (module->data) {
279  if ((rc = module->def.on_destroy(module->data)) != HPD_E_SUCCESS)
280  goto module_destroy_error;
281  module->data = NULL;
282  }
283 
284  // Deallocate run-time memory
285  if ((rc = daemon_free_conf(hpd->configuration))) goto return_error;
286  hpd->configuration = NULL;
287 
288  // Return
289  return HPD_E_SUCCESS;
290 
291  // Return in a nice state if an error occur
292  module_start_error:
293  module_stop_error:
294  for (module = TAILQ_PREV(module, modules, HPD_TAILQ_FIELD); module; module = TAILQ_PREV(module, modules, HPD_TAILQ_FIELD))
295  module->def.on_stop(module->data, hpd);
296  HPD_TAILQ_FOREACH_SAFE(async, &hpd->request_watchers, async_tmp) {
297  TAILQ_REMOVE(&hpd->request_watchers, async, HPD_TAILQ_FIELD);
298  ev_async_stop(hpd->loop, &async->watcher);
299  request_free_request(async->request); // TODO Handle error code
300  free(async);
301  }
302  HPD_TAILQ_FOREACH_SAFE(async, &hpd->respond_watchers, async_tmp) {
303  TAILQ_REMOVE(&hpd->respond_watchers, async, HPD_TAILQ_FIELD);
304  ev_async_stop(hpd->loop, &async->watcher);
305  request_free_response(async->response); // TODO Handle error code
306  free(async);
307  }
308  HPD_TAILQ_FOREACH_SAFE(async, &hpd->changed_watchers, async_tmp) {
309  TAILQ_REMOVE(&hpd->changed_watchers, async, HPD_TAILQ_FIELD);
310  ev_async_stop(hpd->loop, &async->watcher);
311  discovery_free_sid(async->service); // TODO Handle error code
312  value_free(async->value); // TODO Handle error code
313  free(async);
314  }
315  HPD_TAILQ_FOREACH_SAFE(async, &hpd->attached_watchers, async_tmp) {
316  TAILQ_REMOVE(&hpd->attached_watchers, async, HPD_TAILQ_FIELD);
317  ev_async_stop(hpd->loop, &async->watcher);
318  discovery_free_did(async->device); // TODO Handle error code
319  free(async);
320  }
321  HPD_TAILQ_FOREACH_SAFE(async, &hpd->detached_watchers, async_tmp) {
322  TAILQ_REMOVE(&hpd->detached_watchers, async, HPD_TAILQ_FIELD);
323  ev_async_stop(hpd->loop, &async->watcher);
324  discovery_free_did(async->device); // TODO Handle error code
325  free(async);
326  }
327  ev_new_error:
328  arg_error:
329  module = TAILQ_LAST(&hpd->modules, modules);
330  module_create_error:
331  module_destroy_error:
332  for (; module; module = TAILQ_PREV(module, modules, HPD_TAILQ_FIELD))
333  if (module->data) module->def.on_destroy(module->data);
334  alloc_error:
335  if (hpd->configuration) {
336  daemon_free_conf(hpd->configuration);
337  hpd->configuration = NULL;
338  }
339  if (hpd->options) {
340  free(hpd->options);
341  hpd->options = NULL;
342  }
343  return_error:
344  switch (rc) {
345  case HPD_E_ALLOC:
346  LOG_RETURN_E_ALLOC();
347  default:
348  return rc;
349  }
350 }
