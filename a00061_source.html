189 #define FIND_ADAPTER(ID, ADAPTER) do { \
190  (ADAPTER) = NULL; \
191  HPD_TAILQ_FOREACH((ADAPTER), &(ID)->hpd->configuration->adapters) \
192  if (strcmp((ADAPTER)->id, (ID)->aid) == 0) break; \
193  if (!(ADAPTER)) return HPD_E_NOT_FOUND; \
194 } while (0)
195 
196 #define FIND_DEVICE(ID, DEVICE) do { \
197  hpd_adapter_t *a = NULL; \
198  FIND_ADAPTER(&(ID)->adapter, a); \
199  (DEVICE) = NULL; \
200  HPD_TAILQ_FOREACH((DEVICE), a->devices) \
201  if (strcmp((DEVICE)->id, (ID)->did) == 0) break; \
202  if (!(DEVICE)) return HPD_E_NOT_FOUND; \
203 } while (0)

205 #define FIND_SERVICE(ID, SERVICE) do { \
206  hpd_device_t *d = NULL; \
207  FIND_DEVICE(&(ID)->device, d); \
208  (SERVICE) = NULL; \
209  HPD_TAILQ_FOREACH((SERVICE), d->services) \
