107 static hpd_error_t hpd_demo_adapter_start(void *data, hpd_t *hpd)
108 {
109  struct hpd_demo_adapter *demo_adapter = data;
110  hpd_error_t rc;
111 
112  printf("[adapter] Starting with %i lamps...\n", demo_adapter->num_lamps); // TODO Wrong!
113 
114  // TODO Clean up on errors !
115 
116  // Create adapter structure
117  // Using module_id as id, because we are only creating one adapter per module in this scenario,
118  // and we don't have a more informal id from the underlying device.
119  hpd_adapter_t *adapter;
120  if ((rc = hpd_adapter_alloc(&adapter, demo_adapter->module_id))) return rc;
121  if ((rc = hpd_adapter_set_attr(adapter, HPD_ATTR_TYPE, "demo_adapter"))) return rc;
122  // Hand it over to hpd
123  // Hereafter the pointer should not be used, only references by id
124  if ((rc = hpd_adapter_attach(hpd, adapter))) return rc;
125 
126  // Create id structure to reference our adapter
127  hpd_adapter_id_alloc(&demo_adapter->adapter_id, hpd, demo_adapter->module_id);
128 
129  // Create device structures
130  for (size_t i = 0; i < demo_adapter->num_lamps; i++) {
131  char *id = NULL;
132  HPD_SPRINTF_ALLOC(id, "dev%zu", i);
133 
134  hpd_device_t *device;
135  if ((rc = hpd_device_alloc(&device, id))) return rc;
136  if ((rc = hpd_device_set_attr(device, HPD_ATTR_TYPE, "demo_lamp"))) return rc;
137  hpd_service_t *service;
138  if ((rc = hpd_service_alloc(&service, "srv0"))) return rc;
139  if ((rc = hpd_service_set_actions(service,
140  HPD_M_GET, hpd_demo_on_get,
141  HPD_M_PUT, hpd_demo_on_put,
142  HPD_M_NONE))) return rc;
143  hpd_parameter_t *parameter;
144  if ((rc = hpd_parameter_alloc(&parameter, "param0"))) return rc;
145  if ((rc = hpd_parameter_attach(service, parameter))) return rc;
146  if ((rc = hpd_service_attach(device, service))) return rc;
147  if ((rc = hpd_device_attach(demo_adapter->adapter_id, device))) return rc;
148 
149  free(id);
150  }
151 
152  return HPD_E_SUCCESS;
153 
154  alloc_error:
155  return HPD_E_ALLOC;
156  nsprintf_error:
157  return HPD_E_UNKNOWN;
158 }
